# disable compilier warnings in machine-generated code
add_compile_options(-w)

# XML

set(
  keeper_xml
  "com.canonical.keeper.xml"
)

set(
  user_xml
  "com.canonical.keeper.User.xml"
)

set_source_files_properties(
    "${keeper_xml}"
    "${user_xml}"
    PROPERTIES
    NO_NAMESPACE YES
    INCLUDE "qdbus-stubs/dbus-types.h"
)

# Service Object

set_source_files_properties(
  ${keeper_xml}
  PROPERTIES
  CLASSNAME DBusInterfaceKeeper
)

qt5_add_dbus_interface(
  interface_files
  ${keeper_xml}
  keeper_interface
)

qt5_add_dbus_adaptor(
  adaptor_files
  ${keeper_xml}
  ${CMAKE_SOURCE_DIR}/src/service/keeper.h
  Keeper
  KeeperAdaptor
)

# User Object

set_source_files_properties(
  ${user_xml}
  PROPERTIES
  CLASSNAME DBusInterfaceKeeperUser
)

qt5_add_dbus_interface(
  interface_files
  ${user_xml}
  keeper_user_interface
)

qt5_add_dbus_adaptor(
  adaptor_files
  ${user_xml}
  ${CMAKE_SOURCE_DIR}/src/service/keeper-user.h
  KeeperUser
  KeeperUserAdaptor
  KeeperUserAdaptor
)

#
#

add_library(
  qdbus-stubs
  STATIC
  ${interface_files}
  ${adaptor_files}
)
